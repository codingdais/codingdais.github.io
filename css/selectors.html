/* all attribute sign meaning*/

/* 
    ^ attribute begins with value
    img[alt^="image"]{}

    | attribute begins with value or is first in a dash separated list
    img[alt|="image"] {} ex. image-of

    $ attribue ends with value
    img[alt$='cat']{} ex. image of cat
    a[href$=".com"] {} www.google.com

    * Value occurs anywhere in attribute mean kahi bhi value ho
    img[alt$='cat']{} ex. image of cat in house

    ~ value matches attribute in space separated list mean kahi bi ho with space
    img[alt$='of']{} ex. image of cat in house
