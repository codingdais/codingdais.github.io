# Vuex

  • State management system for vue, loosely based on the flux type pattern
  • Vuex Modules are a way to deal with namespacing
  • Vuex Helpers make a bit more easy to grab data from your store

Store Folder Index.js

import Vue from "vue";
import Vuex from "vuex";

Vue.use(Vuex);

export default new Vuex.Store({
  // main state anything like price, counter
  state: {
    counter: 1,
  },
  // help us access value of state
  getters: {
    counterFun(state) {
      return state.counter;
    },
  },
  // syncronous
  mutations: {
    increament(state, value) {
      state.counter += value;
    },
  },
  // asyncronous
  actions: {
    increamentA(state, value) {
      state.commit("increament", value);
    },
  },
  modules: {},
});

MyVuex.vue

<template>
  <div>
    Counter : {{ $store.getters.counterFun }}
    <!-- syncronous -->
    <button @click="$store.commit('increament', 1)">Increament syncronous</button>
    <!-- asycronous -->
    <button @click="$store.dispatch('increamentA', 1)">Increament asycronous</button>
  </div>
</template>

<script>
export default {
  created() {
      // state show
    console.log(this.$store);
 // show only value
    console.log(this.$store.getters.counterFun);
  }
};
</script>
